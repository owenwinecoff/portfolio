Creative Coding - Perlin

This project uses Perlin noise in p5.js to generate smooth, organic patterns that evolve as the frame is created, creating a dynamic and natural-looking visual composition. It explores the intersection of randomness and control, transforming algorithmic noise into an expressive form of digital art.

Project Code


let numLines = 8;  // how many terrain lines
let numStars = 5; // how many stars
let offsets = [];  // independent noise offsets for each line

function setup() {
  createCanvas(windowWidth, windowHeight);
  background(0);
  noFill();

  // assign each line a unique noise offset
  for (let i = 0; i < numLines; i++) {
    offsets.push(random(1000));
  }
}

function draw() {
  let noiseLevel = height / 3;
  let noiseScale = 0.01;
  let x = frameCount;

  if (x > width) noLoop(); // stop when reaching the right edge

  // draw terrain lines
  for (let i = 0; i < numLines; i++) {
    let nx = noiseScale * x + offsets[i];
    let y = height / 2 + i * 30 + noiseLevel * (noise(nx) - 0.5);

    let r = 150 + 105 * noise(nx + 10);
    let g = 150 + 105 * noise(nx + 20);
    let b = 150 + 105 * noise(nx + 30);

    stroke(r, g, b);
    line(x, height, x, y);
  }

  // draw stars only if above terrain
  for (let i = 0; i < numStars; i++) {
    let starX = random(width);
    let starY = random(height);
    let terrainY = getTerrainTopAt(starX, noiseLevel, noiseScale); // find top terrain height here
    if (starY + 10 < terrainY) {
      let starSize = 1 + 2 * noise(0.1 * starX, 0.1 * starY, frameCount * 0.01);
      fill(255);
      noStroke();
      ellipse(starX, starY, starSize);
    }
  }
}



// helper: iterate through lines to find the top terrain height at x
function getTerrainTopAt(x, noiseLevel, noiseScale) {
  let minY = height;
  for (let i = 0; i < numLines; i++) {
    let nx = noiseScale * x + offsets[i];
    let y = height/2 + i * 30 + noiseLevel * (noise(nx) - 0.5);
    if (y < minY) {
      minY = y;
    }
  }
  return minY;
}
